```tsx
// VORHER (innerhalb des IIFE-Blocks):
```

```tsx
function SortableSection({ id, children }: { id: string; children: React.ReactNode }) {
  const { attributes, listeners, setNodeRef, transform, transition, isDragging } = useSortable({ id })
  return (
    <div
      ref={setNodeRef}
      style={{ transform: CSS.Transform.toString(transform), transition }}
      className={cn('relative', isDragging && 'z-10 opacity-80')}
    >
      {/* Drag-Handle — nur auf Desktop sichtbar */}
      <div
        {...attributes}
        {...listeners}
        className="hidden md:flex absolute -left-6 top-3 cursor-grab active:cursor-grabbing text-slate-300 hover:text-slate-500 focus:outline-none focus:ring-1 focus:ring-slate-400 rounded"
        aria-label="Abschnitt verschieben"
      >
        ⠿
      </div>
      {children}
    </div>
  )
}
```

- [ ] **Schritt 6: Result-View mit DndContext + SortableContext umhüllen**

Den bestehenden result-view-Block (das äußere `<div className="space-y-6">` oder ähnlich) ersetzen durch:

```tsx
<DndContext
  sensors={sensors}
  collisionDetection={closestCenter}
  onDragEnd={(event: DragEndEvent) => {
    const { active, over } = event
    if (over && active.id !== over.id) {
      setSectionOrder(prev => {
        const oldIndex = prev.indexOf(active.id as SectionId)
        const newIndex = prev.indexOf(over.id as SectionId)
        return arrayMove(prev, oldIndex, newIndex)
      })
    }
  }}
>
  <SortableContext items={sectionOrder} strategy={verticalListSortingStrategy}>
    <div className="space-y-6 pl-0 md:pl-8"> {/* pl-8 für Drag-Handle-Platz auf Desktop */}
      {sectionOrder.map(sectionId => {
        switch (sectionId) {
          case 'cost': return <SortableSection key="cost" id="cost">{/* Kosten-Block */}</SortableSection>
          case 'narrative': return <SortableSection key="narrative" id="narrative">{/* Narrativ-Karte */}</SortableSection>
          case 'decisions': return <SortableSection key="decisions" id="decisions">{/* Key Decisions */}</SortableSection>
          case 'rasic': return <SortableSection key="rasic" id="rasic">{/* RASIC + EAM */}</SortableSection>
          case 'joule': return <SortableSection key="joule" id="joule">{/* Joule Use Cases */}</SortableSection>
          default: return null
        }
      })}
    </div>
  </SortableContext>
</DndContext>
```

**Hinweis:** Verschiebe den existierenden JSX-Inhalt jeder Sektion in das jeweilige `case`. Falls eine Sektion von einer Bedingung abhängt (z.B. `resultAudience !== 'exec'`), behalte die Bedingung innerhalb des `SortableSection`-Blocks.
